\begin{align*} \abs{\int_{\partial T^n} f(z)\,dz} \amp \le \int_{\partial T^n} \abs{f(z) - f(z_0) - (z-z_0)f'(z_0)}\,dz \\ \amp \le \int_{\partial T^n} \epsilon \abs{z - z_0} \,dz \\ \amp \le \operatorname{length}(\partial T^n) \cdot \epsilon \sup_{z\in \partial T^n} \abs{z - z_0} \\ \amp \le \epsilon \operatorname{length}(\partial T^n)^2 \end{align*}
