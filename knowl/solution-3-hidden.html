\begin{align} \frac{f(z)}{g(z)} \amp = \frac{c_m (z-a)^m + c_{m+1}(z-a)^{m+1} + \cdots}{d_n (z-a)^n + d_{n+1}(z-a)^{n+1} + \cdots}\notag\\ \amp = \frac{1}{(z-a)^{n-m}} \left( \frac{c_m(z-a)^m + c_{m+1}(z-a)^{m+1} + \cdots}{d_n (z-a)^m + d_{n+1}(z-a)^{m+1} + \cdots} \right)\notag\\ \amp = \frac{1}{(z-a)^{n-m}} \left( \frac{c_m + c_{m+1}(z-a) + \cdots}{d_n + d_{n+1}(z-a) + \cdots} \right)\tag{5.2.2} \end{align}
Notice that
\begin{equation*} \frac{1}{(z-a)^{n-m}} = \begin {cases} \frac{1}{(z-a)^{n-m}} \amp \text{if } m \lt n \\ (z-a)^{m-n} \amp \text{if } m \gt n \end {cases}, \end{equation*}
and that the second factor of (5.2.2) is nonzero at \(z=a\text{.}\)
